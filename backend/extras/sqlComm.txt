CREATE TABLE users (
  id INT NOT NULL AUTO_INCREMENT,
  email VARCHAR(255) NOT NULL UNIQUE,
  password VARCHAR(255) NOT NULL,
  PRIMARY KEY (id)
);
CREATE TABLE books (
  id INT NOT NULL AUTO_INCREMENT,
  title VARCHAR(45) NOT NULL,
  desc VARCHAR(255) NOT NULL,
  cover VARCHAR(45),
  price VARCHAR(45) NOT NULL,
  status VARCHAR(10),
  PRIMARY KEY (id)
);
